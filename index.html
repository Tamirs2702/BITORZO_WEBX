Linux (Debian-based) Install Instructions 

curl -s https://updates.Bitorzo.com/desktop/apt/keys.asc | sudo apt-key add -
echo "deb [arch=amd64] https://updates.Bitorzo.com/desktop/apt xenial main" | sudo tee -a /etc/apt/sources.list.d/Bitorzo-xenial.list
sudo apt update && sudo apt install Bitorzo -desktop
